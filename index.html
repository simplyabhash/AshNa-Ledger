function jsonResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function getSheet(name) { return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name); } function sheetToObjects(sheet) { var data = sheet.getDataRange().getValues(); if (data.length < 2) return []; var headers = data[0]; var rows = data.slice(1).filter(function(r){ return r.join('') !== ''; }); return rows.map(function(row) { var obj = {}; headers.forEach(function(h, i) { obj[h] = row[i]; }); return obj; }); } function forceTextValue(headerName, value){ if ((headerName === 'date' || headerName === 'monthKey') && typeof value === 'string' && value.length > 0 && value.charAt(0) !== "'") { return "'" + value; } return value; } function appendRow(sheet, obj, headers) { var row = headers.map(function(h) { return forceTextValue(h, obj[h] !== undefined ? obj[h] : ''); }); sheet.appendRow(row); } function bulkAppendRows(sheet, objectsArray, headers) { if (objectsArray.length === 0) return; var rows = objectsArray.map(function(obj) { return headers.map(function(h) { return forceTextValue(h, obj[h] !== undefined ? obj[h] : ''); }); }); var startRow = sheet.getLastRow() + 1; sheet.getRange(startRow, 1, rows.length, headers.length).setValues(rows); } function findRowIndexByKey(sheet, keyCol, keyVal) { var data = sheet.getDataRange().getValues(); var headers = data[0]; var colIdx = headers.indexOf(keyCol); for (var i = 1; i < data.length; i++) { if (String(data[i][colIdx]) === String(keyVal)) return i + 1; } return -1; } function updateRow(sheet, keyCol, keyVal, obj) { var rowIdx = findRowIndexByKey(sheet, keyCol, keyVal); if (rowIdx === -1) return false; var headers = sheet.getDataRange().getValues()[0]; var row = headers.map(function(h) { return forceTextValue(h, obj[h] !== undefined ? obj[h] : ''); }); sheet.getRange(rowIdx, 1, 1, row.length).setValues([row]); return true; } function deleteRow(sheet, keyCol, keyVal) { var rowIdx = findRowIndexByKey(sheet, keyCol, keyVal); if (rowIdx === -1) return false; sheet.deleteRow(rowIdx); return true; } function replaceSheetData(sheet, objects, headers) { var lastRow = sheet.getLastRow(); if (lastRow > 1) sheet.getRange(2, 1, lastRow - 1, headers.length).clearContent(); if (objects.length === 0) return; var rows = objects.map(function(obj) { return headers.map(function(h) { return forceTextValue(h, obj[h] !== undefined ? obj[h] : ''); }); }); sheet.getRange(2, 1, rows.length, headers.length).setValues(rows); } function setupFormatting(){ var exp = getSheet('Expenses'); exp.getRange('B2:B').setNumberFormat('@'); exp.getRange('K2:K').setNumberFormat('@'); var posted = getSheet('Posted'); posted.getRange('B2:B').setNumberFormat('@'); } var EXPENSE_HEADERS = ['id','date','category','amount','paidBy','mode','remarks','source','createdAt','isRefund','refundOf','card']; var TEMPLATE_HEADERS = ['id','name','category','amount','paidBy','mode','card']; var CARD_HEADERS = ['id','name','cycleDay','dueOffsetDays']; function doGet(e) { var action = (e.parameter && e.parameter.action) || 'getAll'; if (action === 'getAll') { return jsonResponse({ expenses: sheetToObjects(getSheet('Expenses')), templates: sheetToObjects(getSheet('Templates')), posted: sheetToObjects(getSheet('Posted')), categories: sheetToObjects(getSheet('Categories')), cards: sheetToObjects(getSheet('Cards')) }); } return jsonResponse({ error: 'unknown action' }); } function doPost(e) { var body = JSON.parse(e.postData.contents); var action = body.action; var payload = body.payload; if (action === 'ping') { return jsonResponse({ success: true, pong: true, receivedPayload: payload || null }); } else if (action === 'addExpense') { appendRow(getSheet('Expenses'), payload, EXPENSE_HEADERS); } else if (action === 'bulkAddExpenses') { bulkAppendRows(getSheet('Expenses'), payload, EXPENSE_HEADERS); } else if (action === 'updateExpense') { updateRow(getSheet('Expenses'), 'id', payload.id, payload); } else if (action === 'deleteExpense') { deleteRow(getSheet('Expenses'), 'id', payload.id); } else if (action === 'setTemplates') { replaceSheetData(getSheet('Templates'), payload, TEMPLATE_HEADERS); } else if (action === 'setPosted') { replaceSheetData(getSheet('Posted'), payload, ['templateId','monthKey']); } else if (action === 'setCategories') { replaceSheetData(getSheet('Categories'), payload, ['name']); } else if (action === 'setCards') { replaceSheetData(getSheet('Cards'), payload, CARD_HEADERS); } else { return jsonResponse({ success: false, error: 'unknown action' }); } return jsonResponse({ success: true }); }
